24  int clkMode=mode;

33  data=SIXTEEN_BITS;

35  }

37  LPC_SPI->SPCR |= data<<BITS_BIT_POSITION; //set new value

46 int SPI_Transfer(unsigned short *txBuffer, unsigned short *rxBuffer, int length){
47  short status=LPC_SPI->SPSR; //clear the status register. Optional but suggested by the manufacturer
48  status=SPIF;
49  for(int i=0; i<length && status==SPIF;i++){
50  status=0; //clear the status of the previous transfer
51  LPC_SPI->SPDR=txBuffer[i];
52  do{
53  status=((LPC_SPI->SPSR)>>3)&0x1F;
54  }while(status==0);
55  if(status==MODF)
56  LPC_SPI->SPCR |= 1<<5;
57  else if(status==WCOL)
58  LPC_SPI->SPDR;
59  else if(status==SPIF)
60  rxBuffer[i]=LPC_SPI->SPDR;
61  }
62  return status;
63 }
